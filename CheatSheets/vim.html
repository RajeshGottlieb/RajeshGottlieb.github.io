Vim Cheat Sheet

line numbers

:set number
:set nonumber

# Toggle NERDTree
Ctrl-o

windows

:help windows

# split horizontal
CTRL-w s

# split vertical
CTRL-w s

# toggle (cycle) to other windows
CTRL-w w

# move to left window
CTRL-w h

# move to right window
CTRL-w h

# move to upper window
CTRL-w k

# move to lower window
CTRL-w j

# close current window
:bd

vimdiff

:help diff

# Ignore white space in vimdiff
:set diffopt+=iwhite

# switch between left and right windows
Ctrl-w w

# Jump backwards to previous diff
[c

# Jump forwards to next diff
]c

# obtain diff from other window
do

# put diff into other window
dp

:help fold

# Reduce folding
zr

# fold more
zm

# Open one fold under the cursor
zo

# Close one fold under the cursor
zc
